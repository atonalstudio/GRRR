.my-container.grrr {
  --grrr-cols:      12;
  --grrr-gutter:   18px;
  --grrr-margin:   24px;
  --grrr-col-width: 84px;

  /* optional named area shortcuts */
  --content-area: col-start 4 / col-end 9;
  --wide-area:    col-start 2 / col-end 11;

.my-component {
